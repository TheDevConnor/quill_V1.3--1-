func async calculateTheAreaOfATriangleSSS (a, b, c){
    # Calculate the area
    const CalculateArea := {
        area: sqrt((a + b + c) * (b + c - a)
            * (c + a - b) * (a + b - c)) / 4,
    }
    
    # Print out the results of the calculations
    trace(CalculateArea[area])
    trace(a, b, c)
}
